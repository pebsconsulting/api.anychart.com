 *      "hAlign": smth,
 *      "textWrap": smth,
 *      "textOverflow": smth,
 *      "selectable": smth,
 *      "useHtml": smth
 *    }</code>.
 * @return {anychart.core.Text} Self instance for method chaining.
 */
anychart.core.Text.prototype.textSettings;

/**
 * Setter for the text appearance settings.
 * @detailed Overrides text setting Value by its Name.
